program TestFibonacci;
main {
    int arr[1000];
    arr[0] = 0;
    arr[1] = 1;

    int i;
    for (i = 2; i < 1000; i = i + 1) {
        arr[i] = arr[i - 1] + arr[i - 2];
    };

    int n;
    print("Input N:");
    input(n);
    while (n > 0) {
        print("N = " . n);
        print("arr[N] = " . arr[n]);
        print("Input N:");
        input(n);
    };
}
